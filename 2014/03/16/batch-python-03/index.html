windows server2008上のFileMakerServerのバッチ処理を行うにあたり、作業用csvファイルから動的データを

取得し、コマンドを叩く。

#!/usr/bin/env python
# -*- coding: utf-8 -*-

import os,csv,json

def get_conf(path):
    try:
        os.path.isfile(path)
        f = open(path,'r')
        data = json.load(f)
        f.close()
        return data

    except Exception as e:
        #print "FileNotFound or Access Error."
        return None

def read_file(path):
    try:
        os.path.isfile(path)
        f = open(path,'r')
        data = csv.reader(f)
        users = [line[0] for line in data]
        f.close()
        return users
    except Exception as e:
        #print "FileNotFound or Access Error."
        return None

def bu_cmd(users,bu_path):
    if isinstance(users,list) and len(users) > 0:
        cmds = map(lambda x:"fmsadmin -u account -p password backup %s -d %s -k 7" % (x,bu_path)  ,users)
        for c in cmds:
            os.system(c)
        return 0
    else:
        #データが正常に取得出来なかった
        return None

if __name__ == '__main__':
    conf_data = get_conf("conf.json")
    users = read_file(conf_data["test"]["csvPath"])
    bu_cmd(users,conf_data["test"]["backupPath"])
