今回は、バブルソート

PHPでは、

<?php 
$num = trim(fgets(STDIN));

$num = ($num)?(int)$num:10;

$arr = array(1);
for($i=0;$i<$num;$i++){

    $len = count($arr);
    $tmp = array(1);
    for($j=0;$j<$len-1;$j++){
        array_push($tmp,$arr[$j]+$arr[$j+1]);
    }
    array_push($tmp,1);
    $arr = $tmp;
    echo join(',',$arr)."\n";
}
?>

Perlでは、

#!/usr/bin/perl

use strict;
use warnings;

my $num = <STDIN>;
chomp($num);

$num ||= 10;

my @list = (1);

for(my $i=0;$i<$num;$i++){
    my $len = @list;
    my @tmp = (1);
    for(my $j=0;$j<$len-1;$j++){
        push(@tmp,$list[$j]+$list[$j+1]);
    }
    push(@tmp,1);
    @list = @tmp;
    print join(',',@list)."\n";
}
